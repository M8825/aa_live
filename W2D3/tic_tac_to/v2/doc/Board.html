    class Board

    The Board class is responsible for adding a player’s marks, checking for winners and printing the game board. It has attribute to represent the grid. For simplicity grid is 3x3

def place_mark(position, mark)
  if valid?(position) && empty?(position)
    @grid[position[0]][position[1]] = mark
  else
    raise RuntimeError.new("invalid Mark")
  end

def valid?(position)
  row = position[0]
  col = position[1]

  @grid[row] != nil && @grid[row][col] != nil && (row >= 0 && col >= 0)
end

def win_col?(mark)
  @grid.transpose.each do |row|
    return true if row.all? { |ele| ele == mark }
  end

  return false
end

def win_diagonal?(mark)
  num_of_rows = @grid[0].length
  p, s = 0, 0

  (0...num_of_rows).each do |i|
    p += 1 if @grid[i][i] == mark
    s += 1 if @grid[i][num_of_rows - 1 - i] == mark
  end

  return p == @grid.length || s == @grid.length
end
